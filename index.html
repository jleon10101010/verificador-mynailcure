        async function verificar() {

            const btn = document.querySelector('button');

            const res = document.getElementById('res');

            const url = "TU_URL_DE_GOOGLE_SCRIPT"; // Aquí pondrás tu link de Apps Script

            btn.innerText = "BUSCANDO...";

            try {

                const r = await fetch(`${url}?mfg=${document.getElementById('mfg').value}&exp=${document.getElementById('exp').value}&batch=${document.getElementById('batch').value}`);

                const d = await r.json();

                res.style.display = "block";

                res.style.backgroundColor = d.status === "success" ? "#e8f5e9" : "#ffebee";

                res.innerHTML = d.message;

            } catch (e) { alert("Error de conexión"); }

            btn.innerText = "VERIFICAR ORIGINALIDAD";

        }
